나의 코드
```swift
import Foundation

func solution(_ a:Int, _ b:Int) -> Int {
    
    var answer : Int = 0
    
    answer = Int(String(a) + String(b))! > 2*a*b ? Int(String(a) + String(b))! : 2*a*b
    
    return answer
}
```

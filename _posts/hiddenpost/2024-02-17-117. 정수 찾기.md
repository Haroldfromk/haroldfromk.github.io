```swift
import Foundation

func solution(_ num_list:[Int], _ n:Int) -> Int {
    
    var answer : Int = 0
    
    answer = num_list.contains(n) ? 1 : 0
     
    return answer
}
```
---
배열안에 특정한 값을 포함하는지 물어보는 contains를 사용하였고, 삼항연산자를 통해 결과값을 리턴하였다.

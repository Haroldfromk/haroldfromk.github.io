나의 코드
```swift
import Foundation

let a = Int(readLine()!)!

var str : String = ""

if a % 2 == 0 {
    str = "is even"
} else {
    str = "is odd"
}

print("\(a) \(str)")

```
이렇게 했지만 삼항연산자로도 될것같다.

```swift
import Foundation

let a = Int(readLine()!)!

var str : String = ""

str = a % 2 == 0 ? "is even" : "is odd"

print("\(a) \(str)")
```

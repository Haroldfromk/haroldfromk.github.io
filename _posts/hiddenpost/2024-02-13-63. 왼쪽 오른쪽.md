```swift
import Foundation

func solution(_ str_list:[String]) -> [String] {
    
    var answer : [String] = []
    
    for i in str_list.indices {
        if str_list.count == 1 {
            answer = []
        } else {
            if str_list[i] == "l" {
            
            answer = Array(str_list[..<i])
            break
            
        } else if str_list[i] == "r" {
            answer = Array(str_list[(i+1)...])
            break
            
        }
        }
        
        
    }
    
    return answer
}
```

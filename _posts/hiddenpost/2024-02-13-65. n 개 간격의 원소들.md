```swift
import Foundation

func solution(_ num_list:[Int], _ n:Int) -> [Int] {
    
    var answer : [Int] = []

    answer = stride(from:0, through:num_list.count-1, by:n).map{num_list[$0]}
    
    
    return answer
}
```
---
stride를 사용하여 간격에 따라서 출력을 하려고 하였다.

num_list[] 인데 num_list{}로 하여 에러가 났었다.

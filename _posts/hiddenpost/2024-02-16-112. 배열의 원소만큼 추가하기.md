```swift
import Foundation

func solution(_ arr:[Int]) -> [Int] {
    
    var answer : [Int] = []
    
    for i in arr.indices{
        var n = 1
        while arr[i] >= n {
            answer.append(arr[i])
            n+=1
        }
    }
    
    return answer
}
```
---
for문과 while문을 합쳐서 배열에 arr[i]만큼 append를 반복시켜서 추가하였다.

```swift
import Foundation

func solution(_ a:Int, _ b:Int) -> Int {
    
    var answer : Int = 0
    
    answer = a*b % 2 != 0 ? a*a + b*b : (a%2 == 0 && b%2 == 0) ? abs(a-b) : 2*(a+b)
                                                                      
    return answer
}
```
---
삼항연산자를 이용하여 문제를 풀려고하였다.

모두 홀수가 아닐때 즉 둘다 짝수일때 절대값을 표현해야하는데 abs()를 사용하여 절대값을 표현하였다.

```swift
import Foundation

func solution(_ rsp:String) -> String {

    var answer : String = ""
    var arr : [String] = []

    arr = rsp.map{String($0)}

    for i in arr.indices{
        switch arr[i] {
            case "2" : answer += "0"
            case "0" : answer += "5"
            case "5" : answer += "2"
            default : answer = ""
        }
    }

    return answer
}
```
---

삼항연산자 map 모두 생각했는데 시도를 안해봤다

그냥 for문에 switch case로 해결을 했는데 다음엔 다르게 해봐야겠다.

### 다른코드
```swift
import Foundation

func solution(_ rsp:String) -> String {
    return rsp.map{$0 == "0" ? "5" : $0 == "2" ? "0" : "2"}.joined()
}
```

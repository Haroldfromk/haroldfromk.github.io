func setUIlabel () {
        let label = UILabel()
        label.text = "Hello World!" // Labelм—җ ліҙм—¬мЈјкё°
        label.font = UIFont.systemFont(ofSize: 15) // Font size м„Өм •
        label.textColor = UIColor.red // Font color м„Өм •
        label.textAlignment = .center // Text alignment м„Өм •
        label.numberOfLines = 2 // Number of lines мөңлҢҖ 2мӨ„к№Ңм§Җ н‘ңмӢң
                                // 0мңјлЎң н•ҳл©ҙ мһҗлҸҷмңјлЎң мӨ„мқ„ л°”кҫјлӢӨ.
        label.lineBreakMode = .byTruncatingTail // Textк°Җ л„Ҳл¬ҙ кёёл©ҙ ... мңјлЎң н‘ңмӢң
        
        label.frame = CGRect(x:150, y:150, width: 150, height: 150)
        // мғқм„ұлҗ  labelмқҳ мң„м№ҳмҷҖ нҒ¬кё°лҘј м„ёнҢ…
        label.backgroundColor = #colorLiteral(red: 0.1019607857, green: 0.2784313858, blue: 0.400000006, alpha: 1) // label л°°кІҪмғүмқ„ м„Өм •.
        
        label.layer.borderWidth = 3 // labelм—җ н…Ңл‘җлҰ¬ көөкё°лҘј м„Өм •н•ңлӢӨ.
        label.layer.borderColor = UIColor.red.cgColor // label н…Ңл‘җлҰ¬ мғүмқ„ м„Өм •
                                                                // л’Өм—җ cgColorк°Җ мҳӨлҠ”кұё лӘ…мӢ¬н•ҳмһҗ
        
        self.view.addSubview(label)
        // мӢӨм ңлЎң н•ҙлӢ№ л·°лҘј лқ„мҡ°кІ лӢӨ. мқҙкұё мһ‘м„ұн•ҳм§Җ м•Ҡмңјл©ҙ н•ЁмҲҳлҘј нҳём¶ңн•ҙлҸ„ ліҙмқҙм§Җ м•ҠлҠ”лӢӨ.
    }

func setImageView () {
        let imageView = UIImageView()
        let image = UIImage(named: "test") // Image File лӘ…мңјлЎң UIImage мҳӨлёҢм қнҠё мғқм„ұ
        imageView.image = image
        
        // мқҙлҜём§Җ л·°мқҳ н”„л Ҳмһ„ м„Өм •
        imageView.frame = CGRect(x: 150, y: 350, width: 150, height: 150)
        
        // contentMode м„Өм •
        imageView.contentMode = .scaleAspectFit // мқҙлҜём§Җ л№„мңЁ мң м§Җн•ҳл©ҙм„ң л§һм¶ҳлӢӨ.
        
        // л·°лҘј нҷ”л©ҙм—җ н‘ңмӢң
        self.view.addSubview(imageView)
    }

func setUITextField() {
        let textField = UITextField()
        
        textField.placeholder = "м—¬кё°м—җ мһ…л Ҙн•ҙ"
        textField.borderStyle = .roundedRect // TextFieldмқҳ лӘЁм–‘ (л‘Ҙк·ј н…Ңл‘җлҰ¬)
        textField.keyboardType = .default // мқјл°ҳ нӮӨліҙл“ң
        textField.isSecureTextEntry = false // л№„л°ҖлІҲнҳёмІҳлҹј *** мқҙлҹ°мӢқмңјлЎң к°ҖлҰҙкұҙм§Җмқҳ м—¬л¶Җ
        textField.returnKeyType = .done // ReturnнӮӨмқҳ мў…лҘҳ

        
        textField.frame = CGRect(x: 150, y: 400, width: 150, height: 40)
        
        self.view.addSubview(textField)
        
    }

 func setUIButton() {
        let uiButton = UIButton(type: .system)
        
        uiButton.setTitle("лҲҢлҹ¬лҙҗ", for: .normal) // нҳ„мһ¬ лІ„нҠјмқҳ text
        uiButton.setTitleColor(UIColor.green, for: .normal) // кёҖмһҗмғү
        uiButton.frame = CGRect(x: 150, y: 500, width: 150, height: 50)
        uiButton.backgroundColor = .blue // л°°кІҪмғү
        uiButton.addTarget(self, action: #selector(buttonTapped), for: .touchUpInside) // лІ„нҠјмқ„ лҲҢл Җмқ„л•Ң мӢӨн–үн•  н•ЁмҲҳлҘј м§Җм •н•ҙмӨҖлӢӨ.
        
        self.view.addSubview(uiButton)
    }
    
    @objc func buttonTapped() {
        print("Button Pressed")
    }

func setUISwitch () {
        let uiSwitch = UISwitch()
        uiSwitch.isOn = true
        uiSwitch.onTintColor = .blue
        uiSwitch.thumbTintColor = .orange
        
        uiSwitch.addTarget(self, action: #selector(switchValueChanged(_ :)), for: .valueChanged) // valuechanged : к°’мқҙ ліҖн–Ҳмқ„л•Ң мқём§Җ
        
        uiSwitch.frame = CGRect(x: 50, y: 70, width: 40, height: 40)
        
        self.view.addSubview(uiSwitch)
    }
    
    @objc func switchValueChanged (_ sender : UISwitch ) {
        if sender.isOn {
            print("switch on")
        } else {
            print("switch off")
        }
    }
